SQL query analysis — 5 business patterns identified
1. Loan default rate by product type
Risk
-- Default rates across loan products SELECT l.product_type, COUNT(*) AS total_loans, SUM(CASE WHEN l.status = 'Defaulted' THEN 1 ELSE 0 END) AS defaults, ROUND(defaults * 100.0 / total_loans, 2) AS default_rate FROM loans l GROUP BY l.product_type ORDER BY default_rate DESC;

2. Customer segmentation by AUM & activity
Segment
-- Segment by AUM tier & txn frequency SELECT CASE WHEN c.aum >= 5000000 THEN 'HNI' WHEN c.aum >= 1000000 THEN 'Affluent' ELSE 'Mass Market' END AS segment, COUNT(c.customer_id) AS customers, AVG(t.monthly_txns) AS avg_txns, SUM(c.aum) AS total_aum FROM customers c JOIN transactions t ON c.customer_id = t.customer_id GROUP BY segment ORDER BY total_aum DESC;

3. Branch-wise revenue vs cost efficiency
Performance
-- Rank branches by cost-to-income ratio SELECT b.branch_name, b.region, SUM(r.revenue) AS total_revenue, SUM(r.opex) AS total_cost, ROUND(opex * 100.0 / revenue, 1) AS cir_pct, RANK() OVER ( ORDER BY cir_pct ASC ) AS efficiency_rank FROM branches b JOIN revenue_data r ON b.branch_id = r.branch_id GROUP BY b.branch_name, b.region ORDER BY efficiency_rank;

4. Monthly transaction volume trend
Insight
-- YoY monthly txn volume with MoM growth SELECT DATE_TRUNC('month', t.txn_date) AS txn_month, COUNT(t.txn_id) AS total_txns, SUM(t.amount) AS total_value, LAG(SUM(t.amount)) OVER ( ORDER BY txn_month ) AS prev_month, ROUND((total_value - prev_month) * 100.0 / prev_month, 2) AS mom_growth_pct FROM transactions t WHERE t.txn_date >= '2024-04-01' GROUP BY txn_month ORDER BY txn_month;

5. Early warning — high-risk loan accounts (5-table multi-join)
Risk
-- Flag accounts: overdue + low score + high utilisation SELECT c.customer_id, c.customer_name, l.product_type, l.outstanding_amount, cr.credit_score, u.utilisation_pct, p.days_overdue,
CASE WHEN cr.credit_score < 600 AND u.utilisation_pct > 80 AND p.days_overdue > 30 THEN 'Critical' WHEN cr.credit_score < 650 AND p.days_overdue > 15 THEN 'Watch' ELSE 'Monitor' END AS risk_flag FROM customers c JOIN loans l ON c.customer_id = l.customer_id JOIN credit_scores cr ON c.customer_id = cr.customer_id JOIN utilisation u ON l.loan_id = u.loan_id JOIN payments p ON l.loan_id = p.loan_id WHERE risk_flag != 'Monitor' ORDER BY p.days_overdue DESC LIMIT 5;
